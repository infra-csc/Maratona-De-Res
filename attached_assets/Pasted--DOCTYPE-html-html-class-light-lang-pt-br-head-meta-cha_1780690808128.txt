<script>
        function toggleStatus(btn) {
            btn.classList.toggle('active');
            const dot = btn.querySelector('span');
            const row = btn.closest('.grid');
            if(btn.classList.contains('active')) {
                dot.classList.add('translate-x-8');
                dot.classList.remove('translate-x-1');
                row.classList.remove('opacity-60');
            } else {
                dot.classList.remove('translate-x-8');
                dot.classList.add('translate-x-1');
                row.classList.add('opacity-60');
            }
            updateTotal();
        }

        function updateTotal() {
            const inputs = document.querySelectorAll('.weight-input');
            let total = 0;
            inputs.forEach(input => {
                const row = input.closest('.grid');
                const btn = row.querySelector('.status-toggle');
                if(btn.classList.contains('active')) {
                    total += parseInt(input.value) || 0;
                }
            });

            const display = document.getElementById('totalWeight');
            const track = document.getElementById('weightTrack');
            const warning = document.getElementById('weightWarning');
            
            display.innerText = total;
            const percentage = Math.min((total / 20) * 100, 100);
            track.style.width = percentage + '%';

            if(total === 20) {
                warning.innerText = "META ATINGIDA! PESO IDEAL";
                warning.className = "font-label-bold text-[10px] mt-2 text-primary italic";
                track.classList.replace('bg-primary-container', 'bg-primary-container');
            } else if (total > 20) {
                warning.innerText = "LIMITE EXCEDIDO (" + (total - 20) + " pts acima)";
                warning.className = "font-label-bold text-[10px] mt-2 text-error italic";
                track.classList.replace('bg-primary-container', 'bg-error');
            } else {
                warning.innerText = "FALTAM " + (20 - total) + " PONTOS PARA A META";
                warning.className = "font-label-bold text-[10px] mt-2 text-secondary italic";
                track.classList.replace('bg-error', 'bg-primary-container');
            }
        }

        // Initialize inputs
        document.querySelectorAll('.weight-input').forEach(input => {
            input.addEventListener('input', updateTotal);
        });
        
        // Initial run
        updateTotal();
    </script>
