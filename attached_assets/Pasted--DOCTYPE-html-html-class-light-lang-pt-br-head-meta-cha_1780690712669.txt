<script>
        function setScore(btn, score) {
            // Get siblings in the same row
            const row = btn.closest('.grid');
            const buttons = row.querySelectorAll('.score-btn');
            const criterionRow = btn.closest('.criterion-row');
            const commentBox = criterionRow.querySelector('.comment-box');
            
            // Remove active classes
            buttons.forEach(b => b.classList.remove('score-active'));
            
            // Set active
            btn.classList.add('score-active');
            
            // Handle mandatory comments for score < 3
            if (score < 3) {
                commentBox.classList.remove('hidden');
                criterionRow.style.borderColor = '#ba1a1a'; // error color
            } else {
                commentBox.classList.add('hidden');
                criterionRow.style.borderColor = '#ccff00'; // primary-container
            }
        }

        function selectEvent(btn) {
            // Visual feedback for selection
            const buttons = btn.parentElement.querySelectorAll('button');
            buttons.forEach(b => {
                b.classList.remove('bg-primary-container/20', 'border-primary', 'shadow-[2px_2px_0px_0px_rgba(80,102,0,1)]');
                const check = b.querySelector('.material-symbols-outlined');
                if(check && check.innerText === 'check_circle') check.remove();
                b.querySelector('span.material-symbols-outlined')?.classList.remove('opacity-0');
            });

            btn.classList.add('bg-primary-container/20', 'border-primary', 'shadow-[2px_2px_0px_0px_rgba(80,102,0,1)]');
            
            const checkIcon = document.createElement('span');
            checkIcon.className = 'material-symbols-outlined text-primary';
            checkIcon.innerText = 'check_circle';
            checkIcon.style.fontVariationSettings = "'FILL' 1";
            
            const arrowIcon = btn.querySelector('[data-icon="chevron_right"]');
            if(arrowIcon) arrowIcon.classList.add('opacity-0');
            btn.appendChild(checkIcon);
        }
    </script>
